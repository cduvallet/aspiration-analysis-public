## Do PCA
pca2016 = PCA(random_state=random_state)\
    .fit(jsd.loc[keep_batch, keep_batch].values)

Y2016 = pca2016.transform(jsd.loc[keep_batch, keep_batch].values)

# Put PC's in a df and add metadata
